# Clonar repositório
    - git clone
# Abrir um terminal cmd/powershell no local onde foi clonado e rodar o seguinte comando:
    - docker-compose up -d
# Após receber a mensagem "Example app listening on port 3000" no terminal
# Abra o navegador na URL 
    - http://localhost:3000/
